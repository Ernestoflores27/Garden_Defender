66 void Detector::detect()
67 {
68  real_time.read(frame);
69  cv::Mat blob;
70  cv::dnn::blobFromImage(frame, blob, 1 / 255.0, cv::Size(this->inpWidth, this->inpHeight));
71  this->net.setInput(blob);
72  std::vector<cv::Mat> outs;
73  this->net.forward(outs, this->net.getUnconnectedOutLayersNames());
74 
75  // generate proposals
76  std::vector<int> classIds;
77  std::vector<float> confidences;
78  std::vector<cv::Rect> boxes;
79  float ratioh = (float)frame.rows / this->inpHeight, ratiow = (float)frame.cols / this->inpWidth;
80  int n = 0, q = 0, i = 0, j = 0, nout = this->anchor_num * 5 + this->classes.size(), row_ind = 0;
81  float *pdata = (float *)outs[0].data;
82  for (n = 0; n < this->num_stage; n++)
83  {
84  int num_grid_x = (int)(this->inpWidth / this->stride[n]);
85  int num_grid_y = (int)(this->inpHeight / this->stride[n]);
86  for (i = 0; i < num_grid_y; i++)
87  {
88  for (j = 0; j < num_grid_x; j++)
89  {
90  cv::Mat scores = outs[0].row(row_ind).colRange(this->anchor_num * 5, outs[0].cols);
91  cv::Point classIdPoint;
92  double max_class_socre;
93  // Get the value and location of the maximum score
94  cv::minMaxLoc(scores, 0, &max_class_socre, 0, &classIdPoint);
95  for (q = 0; q < this->anchor_num; q++)
96  {
97  const float anchor_w = this->anchors[n][q * 2];
98  const float anchor_h = this->anchors[n][q * 2 + 1];
99  float box_score = pdata[4 * this->anchor_num + q];
100  if (box_score > this->objThreshold && max_class_socre > this->confThreshold)
101  {
102  float cx = (pdata[4 * q] * 2.f - 0.5f + j) * this->stride[n];
103  float cy = (pdata[4 * q + 1] * 2.f - 0.5f + i) * this->stride[n];
104  float w = powf(pdata[4 * q + 2] * 2.f, 2.f) * anchor_w;
105  float h = powf(pdata[4 * q + 3] * 2.f, 2.f) * anchor_h;
106 
107  int left = (cx - 0.5 * w) * ratiow;
108  int top = (cy - 0.5 * h) * ratioh;
109 
110  classIds.push_back(classIdPoint.x);
111  confidences.push_back(box_score * max_class_socre);
112  boxes.push_back(cv::Rect(left, top, (int)(w * ratiow), (int)(h * ratioh)));
113  // TODO: Add variable in if statement to variable to check status
114  }
115  }
116  row_ind++;
117  pdata += nout;
118  }
119  }
120  }
121 
122  // Perform non maximum suppression to eliminate redundant overlapping boxes with lower confidences
123  objs_vector.clear();
124  std::vector<int> indices;
125  cv::dnn::NMSBoxes(boxes, confidences, this->confThreshold, this->nmsThreshold, indices);
126  for (size_t i = 0; i < indices.size(); ++i)
127  {
128  int idx = indices[i];
129 
130  if (classIds[idx] == 0)
131  {
132  cv::Rect box = boxes[idx];
133  this->drawPred(classIds[idx], confidences[idx], box.x, box.y,
134  box.x + box.width, box.y + box.height, frame);
135  cv::Size s = frame.size();
136  Object f(box, s.width, s.height);
137  this->objs_vector.push_back(f);
138  }
139  }
140  this->sortObjs();
141  this->show();
142  if (manual == false)
143  turretCallback();
144 }
